function updateEnergyDescription() {

const energyLevel = parseInt(document.getElementById('energy-level').value);

const descElement = document.getElementById('energy-description');

const currentHour = new Date().getHours();

const bedtimeHour = parseInt(settings.bedTime.split(':')[0]);

const isBedtime = Math.abs(currentHour - bedtimeHour) <= 2 || currentHour >= 22 || currentHour <= 6;

if (isBedtime) {

descElement.textContent = bedtimeDescriptions[energyLevel];

} else {

descElement.textContent = energyDescriptions[energyLevel];

}

}

function getNextCheckinType() {

const now = new Date();

const currentHour = now.getHours();

const currentMinute = now.getMinutes();

const currentTime = currentHour * 60 + currentMinute;

const wakeTime = parseTime(settings.wakeTime);

const midMorning = wakeTime + (3.5 * 60); // 3.5 hours after wake

const midAfternoon = 15 * 60; // 3:00 PM

const evening = 18 * 60; // 6:00 PM

const bedTime = parseTime(settings.bedTime);

if (currentTime < wakeTime + 30) return "Upon Waking";

if (currentTime < midMorning + 30) return "Mid-Morning Check-in";

if (currentTime < midAfternoon + 30) return "Mid-Afternoon Check-in";

if (currentTime < evening + 30) return "Evening Check-in";

if (currentTime < bedTime + 30) return "Bedtime Check-in";

return "Next Day Check-in";

}

 

function parseTime(timeStr) {

const [hours, minutes] = timeStr.split(':').map(Number);

return hours * 60 + minutes;

}

function saveCheckin() {

const now = new Date();

const energyLevel = parseInt(document.getElementById('energy-level').value);

const sleepQuality = parseInt(document.getElementById('sleep-quality').value);

const sleepDuration = parseFloat(document.getElementById('sleep-duration').value);

const mealType = document.querySelector('input[name="meal-type"]:checked').value;

const activity = document.querySelector('input[name="activity"]:checked').value === 'yes';

const stressLevel = parseInt(document.getElementById('stress-level').value);

const checkin = {

id: Date.now(),

timestamp: now.toISOString(),

energyLevel,

sleepQuality,

sleepDuration,

mealType,

activity,

stressLevel,

weather: {

temp: document.getElementById('weather-temp').textContent,

description: document.getElementById('weather-desc').textContent,

icon: document.getElementById('weather-icon').textContent

}

};

checkins.unshift(checkin); // Add to beginning of array

// Keep only last 30 days of data

const thirtyDaysAgo = new Date();

thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);

checkins = checkins.filter(c => new Date(c.timestamp) > thirtyDaysAgo);

localStorage.setItem('energyCheckins', JSON.stringify(checkins));

// Show success message

showAlert('Check-in saved successfully!', 'success');

// Update analytics

updateAnalytics();

updateHistory();

// Reset form to defaults

resetForm();

}

function updateAnalytics() {

if (checkins.length === 0) {

document.getElementById('avg-energy').textContent = '--';

document.getElementById('peak-time').textContent = '--';

document.getElementById('low-time').textContent = '--';

document.getElementById('sleep-avg').textContent = '--';

return;

}

// Calculate averages

const avgEnergy = (checkins.reduce((sum, c) => sum + c.energyLevel, 0) / checkins.length).toFixed(1);

const avgSleep = (checkins.reduce((sum, c) => sum + c.sleepDuration, 0) / checkins.length).toFixed(1);

// Find peak and low times

const timeEnergy = {};

checkins.forEach(checkin => {

const hour = new Date(checkin.timestamp).getHours();

if (!timeEnergy[hour]) {

timeEnergy[hour] = { total: 0, count: 0 };

}

timeEnergy[hour].total += checkin.energyLevel;

timeEnergy[hour].count += 1;

});

let peakTime = 0, lowTime = 0, peakEnergy = 0, lowEnergy = 10;

Object.keys(timeEnergy).forEach(hour => {

const avg = timeEnergy[hour].total / timeEnergy[hour].count;

if (avg > peakEnergy) {

peakEnergy = avg;

peakTime = hour;

}

if (avg < lowEnergy) {

lowEnergy = avg;

lowTime = hour;

}

});

document.getElementById('avg-energy').textContent = avgEnergy;

document.getElementById('peak-time').textContent = `${peakTime}:00`;

document.getElementById('low-time').textContent = `${lowTime}:00`;

document.getElementById('sleep-avg').textContent = `${avgSleep}h`;

updatePatterns();

drawChart();

}

 

function updatePatterns() {

const patternsList = document.getElementById('patterns-list');

const patterns = [];

if (checkins.length < 3) {

patternsList.innerHTML = '<p>Need more data to identify patterns (minimum 3 check-ins)</p>';

return;

}

// Analyze patterns

const last7Days = checkins.filter(c => {

const checkDate = new Date(c.timestamp);

const weekAgo = new Date();

weekAgo.setDate(weekAgo.getDate() - 7);

return checkDate > weekAgo;

});

if (last7Days.length > 0) {

const avgEnergy = last7Days.reduce((sum, c) => sum + c.energyLevel, 0) / last7Days.length;

const avgSleep = last7Days.reduce((sum, c) => sum + c.sleepDuration, 0) / last7Days.length;

const avgStress = last7Days.reduce((sum, c) => sum + c.stressLevel, 0) / last7Days.length;

if (avgEnergy >= 7) {

patterns.push("✅ Great energy levels this week!");

} else if (avgEnergy <= 4) {

patterns.push("⚠️ Energy levels have been low - consider sleep and stress factors");

}

if (avgSleep < 6) {

patterns.push("😴 Consider increasing sleep duration");

} else if (avgSleep >= 8) {

patterns.push("✅ Excellent sleep duration");

}

if (avgStress > 7) {

patterns.push("😰 High stress levels detected - consider stress management techniques");

}

const activeCheckins = last7Days.filter(c => c.activity).length;

const activityRate = (activeCheckins / last7Days.length) * 100;

if (activityRate > 60) {

patterns.push("🏃‍♂️ Great activity levels!");

} else if (activityRate < 30) {

patterns.push("🚶‍♂️ Consider increasing physical activity");

}

}

if (patterns.length === 0) {

patterns.push("📊 Continue tracking to identify patterns");

}

patternsList.innerHTML = patterns.map(p => `<p>${p}</p>`).join('');

}

 

function drawChart() {

const canvas = document.getElementById('energy-chart');

const ctx = canvas.getContext('2d');

// Clear canvas

ctx.clearRect(0, 0, canvas.width, canvas.height);

if (checkins.length === 0) {

ctx.fillStyle = '#4a5568';

ctx.font = '16px -apple-system, BlinkMacSystemFont, sans-serif';

ctx.textAlign = 'center';

ctx.fillText('No data to display', canvas.width / 2, canvas.height / 2);

return;

}

// Prepare data (last 7 days)

const last7Days = checkins.filter(c => {

const checkDate = new Date(c.timestamp);

const weekAgo = new Date();

weekAgo.setDate(weekAgo.getDate() - 7);

return checkDate > weekAgo;

}).reverse(); // Reverse to show chronological order

if (last7Days.length === 0) return;

// Chart dimensions

const padding = 40;

const chartWidth = canvas.width - (padding * 2);

const chartHeight = canvas.height - (padding * 2);

// Scales

const xStep = chartWidth / Math.max(last7Days.length - 1, 1);

const yScale = chartHeight / 9; // Energy scale 1-10

// Draw grid lines

ctx.strokeStyle = '#e2e8f0';

ctx.lineWidth = 1;

// Horizontal grid lines (energy levels)

for (let i = 1; i <= 10; i++) {

const y = padding + (chartHeight - ((i - 1) * yScale));

ctx.beginPath();

ctx.moveTo(padding, y);

ctx.lineTo(padding + chartWidth, y);

ctx.stroke();

// Labels

ctx.fillStyle = '#4a5568';

ctx.font = '12px -apple-system, BlinkMacSystemFont, sans-serif';

ctx.textAlign = 'right';

ctx.fillText(i.toString(), padding - 10, y + 4);

}

// Draw energy line

ctx.strokeStyle = '#1a365d';

ctx.lineWidth = 3;

ctx.beginPath();

last7Days.forEach((checkin, index) => {

const x = padding + (index * xStep);

const y = padding + (chartHeight - ((checkin.energyLevel - 1) * yScale));

if (index === 0) {

ctx.moveTo(x, y);

} else {

ctx.lineTo(x, y);

}

});

ctx.stroke();

// Draw data points

ctx.fillStyle = '#1a365d';

last7Days.forEach((checkin, index) => {

const x = padding + (index * xStep);

const y = padding + (chartHeight - ((checkin.energyLevel - 1) * yScale));

ctx.beginPath();

ctx.arc(x, y, 4, 0, 2 * Math.PI);

ctx.fill();

});

}

function exportData() {

checkins: checkins,

settings: settings,

exported: new Date().toISOString()

};

const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });

const url = URL.createObjectURL(blob);

const a = document.createElement('a');

a.href = url;

a.download = `energy-tracker-data-${new Date().toISOString().split('T')[0]}.json`;

document.body.appendChild(a);

a.click();

document.body.removeChild(a);

URL.revokeObjectURL(url);

showAlert('Data exported successfully!', 'success');

}
